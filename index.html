The aim of today's class is not to master electronics or embedded systems, but as a stepping stone to explore more.

Activities

4 structured, hands-on activities with space for your classroom diagrams and ready-to-copy example code.

Activity 1: Blinking an LED

While this action seems to be simple, it allows us to test the functionality of the microprocessor as well as showing a simple control flow adopted in the system.

helloworld.ino

      //MEDD 8860 Emerging Technologies in STEM Education
      //Activity 1: Blink
                
      // the setup function runs once when you press reset or power the board
      void setup() {
        // initialize digital pin LED_BUILTIN as an output.
        pinMode(13, OUTPUT);
      }
                
      // the loop function runs over and over again forever
      void loop() {
        digitalWrite(13, HIGH);  // turn the LED on (HIGH is the voltage level)
        delay(1000);                      // wait for a second
        digitalWrite(13, LOW);   // turn the LED off by making the voltage LOW
        delay(1000);                      // wait for a second
      }

Activity 2: Driving a Motor with different Speed

While we have played with a motor before, this is the first time we are able to control the rotation of the motor to a greater extent, which includes the speed and rotational direction of the motor (not for today's example, as the particular type of motor I sourced is for making fans exclusively.)

helloworld.ino

      //MEDD 8860 Emerging Technologies in STEM Education
      //Activity 2: Driving a Motor
                
      void setup() {
        // initialize digital pin LED_BUILTIN as an output.
        pinMode(3, OUTPUT); //Enable Pin for Motor (Speed Control)
        pinMode(4, OUTPUT); //Input1 Pin for Motor (Direction Control)
        pinMode(5, OUTPUT); //Input2 Pin for Motor (Direction Control)
      }
                
      // the loop function runs over and over again forever
      void loop() {
        digitalWrite(4, HIGH);  
        digitalWrite(5, LOW);  
        analogWrite(3,255);
        delay(5000);                      // wait for 5 second
                
        analogWrite(3,100);
        delay(5000); 
                
      }

Activity 3: Reading a sensor

This is a simple example of retrieving readings from a basic ultrasound sensor, which is one of the commonly used inputs for object detection or distance measurement.

helloworld.ino

      //MEDD 8860 Emerging Technologies in STEM Education
      //Activity 3: Reading a sensor
      
      const int trigPin = 9;
      const int echoPin = 10;
      
      float duration, distance;
      
      void setup() {
        pinMode(trigPin, OUTPUT);
        pinMode(echoPin, INPUT);
        Serial.begin(9600);
      }
      
      void loop() {
        digitalWrite(trigPin, LOW);
        delayMicroseconds(2);
        digitalWrite(trigPin, HIGH);
        delayMicroseconds(10);
        digitalWrite(trigPin, LOW);
      
        duration = pulseIn(echoPin, HIGH);
        distance = (duration*.0343)/2;
        Serial.print("Distance: ");
        Serial.println(distance);
        delay(100);
      }

Activity 4: Putting them together

Let's see if you can put everything together to build a circuit in which the fan will turn on when the sensor detects an object nearby.

helloworld.ino

      //MEDD 8860 Emerging Technologies in STEM Education
      //Activity 4: Putting them together
      
      const int trigPin = 9;
      const int echoPin = 10;
      
      float duration, distance;
      
      void setup() {
        pinMode(trigPin, OUTPUT);
        pinMode(echoPin, INPUT);
        Serial.begin(9600);
      
        pinMode(3, OUTPUT); //Enable Pin for Motor (Speed Control)
        pinMode(4, OUTPUT); //Input1 Pin for Motor (Direction Control)
        pinMode(5, OUTPUT); //Input2 Pin for Motor (Direction Control)
      }
      
      void loop() {
        digitalWrite(trigPin, LOW);
        delayMicroseconds(2);
        digitalWrite(trigPin, HIGH);
        delayMicroseconds(10);
        digitalWrite(trigPin, LOW);
      
        duration = pulseIn(echoPin, HIGH);
        distance = (duration*.0343)/2;
        Serial.print("Distance: ");
        Serial.println(distance);
      
        if (distance<10){
          digitalWrite(4, HIGH);  
          digitalWrite(5, LOW);  
          analogWrite(3,255);
        }
        else{
          digitalWrite(4, LOW);  
          digitalWrite(5, LOW);  
          analogWrite(3,0);
        }
        delay(100);
      }
